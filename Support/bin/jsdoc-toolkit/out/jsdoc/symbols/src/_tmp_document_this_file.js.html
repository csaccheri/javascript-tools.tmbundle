 13 /**
 14  * Makes API request for profile.
 15  */
 16 function getProfile() {
 17   var url = 'http://api.linkedin.com/v1/people/~:(first-name,last-name,site-standard-profile-request:(url),picture-url)';
 18   var request = {
 19     'method': 'GET'
 20   };
 21   oauth.sendSignedRequest( url, processProfile, request );
 22 }

 24 /**
 25  * Makes API request for last 25 Network Updates.
 26  */
 27 function getNetworkUpdates() {
 28   var url = 'http://api.linkedin.com/v1/people/~/network';
 29   var request = {
 30     'method': 'GET',
 31     'parameters': {
 32       'type': 'STAT',
 33       'count': '25'
 34     }
 35   };
 36   oauth.sendSignedRequest( url, processNetworkUpdates, request );

 39 /**
 40  * Makes API request for Network Updates since a given timestamp.
 41  * @param {Number} timestamp Timestamp we want to find updates since.
 42  */
 43 function checkForNewUpdates( timestamp ) {
 44   var url = 'http://api.linkedin.com/v1/people/~/network';
 45   var request = {
 46     'method': 'GET',
 47     'parameters': {
 48       'type': 'STAT',
 49       'after': timestamp
 50     }
 51   };
 52   oauth.sendSignedRequest( url, updateBadge, request );
 53 }

 55 /**
 56  * Makes API request to post a status update.
 57  * @param {String} txt The text of the update we want to post.
 58  */
 59 function postStatusUpdate( txt ) {
 60   var url = 'http://api.linkedin.com/v1/people/~/current-status';
 61   var request = {
 62     'method': 'PUT',
 63     'headers': {
 64       'Content-Type': 'text/xml'
 65     },
 66     'body': '<?xml version=\'1.0\' encoding=\'UTF-8\'?><current-status>' + txt + '</current-status>'
 67   };
 68   oauth.sendSignedRequest( url, processStatusUpdate, request, [ txt ] );
 69 }

 71 /**
 72  * Makes API request to post a comment on a Network Update.
 73  * @param {String} txt The text of the comment we want to post.
 74  * @param {String} key Unique identifier for the update we are attaching this
 75  *     comment to.
 76  */
 77 function postComment( txt, key ) {
 78   var url = 'http://api.linkedin.com/v1/people/~/network/updates/key=' + key + '/update-comments';
 79   var request = {
 80     'method': 'POST',
 81     'headers': {
 82       'Content-Type': 'text/xml'
 83     },
 84     'body': '<?xml version=\'1.0\' encoding=\'UTF-8\'?><update-comment><comment>' + txt + '</comment></update-comment>'
 85   };
 86   oauth.sendSignedRequest( url, processComment, request, [ txt, key ] );
 87 }

 89 /**
 90  * Updates count in toolbar badge.
 91  * @param {String} The text response of the API XHR call.
 92  * @param {XMLHttpRequest} resp The XMLHttpRequest object which was used to
 93  *     send the request.
 94  */
 95 function updateBadge( resp, xhr ) {
 96   // parse responseXML because API does not support JSON yet
 97   var xml = xhr.responseXML;
 98   var numNewUpdates = $( xml ).find( 'update' ).length;
 99     
100   if( numNewUpdates > 0 ) {
101     chrome.browserAction.setBadgeText({
102       text: '' + numNewUpdates
103     });
104   }
105   
106   // update lastCheckTime
107   localStorage.lastCheckTime = new Date().getTime();
108 }

110 /**
111  * Processes response to API request for Profile data, storing key pieces of
112  * data in localStorage.
113  * @param {String} The text response of the API XHR call.
114  * @param {XMLHttpRequest} resp The XMLHttpRequest object which was used to 
115  *     send the request.
116  */
117 function processProfile( resp, xhr ) {
118   // parse responseXML because API does not support JSON yet
119   var xml = xhr.responseXML;
120   
121   localStorage.firstName = $( xml ).find( 'first-name' ).text();
122   localStorage.lastName = $( xml ).find( 'last-name' ).text();
123   localStorage.profileUrl = $( xml ).find( 'site-standard-profile-request' ).first().find( 'url' ).text();
124   localStorage.userPhoto = $( xml ).find( 'picture-url' ).text();
125   if( localStorage.userPhoto === '' ) localStorage.userPhoto = DEFAULT_USER_PHOTO;
126 }

128 /**
129  * Processes response to API request for NUS data, including rendering of the
130  * updates and any comments.
131  * @param {String} The text response of the API XHR call.
132  * @param {XMLHttpRequest} resp The XMLHttpRequest object which was used to 
133  *     send the request.
134  */
135 function processNetworkUpdates( resp, xhr ) {
136   // parse responseXML because API does not support JSON yet
137   var xml = xhr.responseXML;  
138 
139   if( $( xml ).find( 'error' ).length > 0 ) {
140     $( '.loading' ).html( 'Uh-oh. Your updates could not be retrieved. Please try again later.' );
141     return;
142   }
143 
144   if( $( xml ).find( 'updates' ).first().attr( 'total' ) === 0 ) {
145     $( '.loading' ).html( 'No updates found.' );
146     $( '<ul id="feed"><li></li></ul>' ).insertAfter( '.loading' );
147   } else {
148     // get time of most recent update and store in local storage
149     var latestTimestamp = $( xml ).find( 'timestamp' ).first().text();
150     if( latestTimestamp > localStorage.mostRecentTimestamp ) {
151       localStorage.mostRecentTimestamp = latestTimestamp;
152     }
153 
154     // create list of updates
155     var html = '<ul id="feed">';
156     $( xml ).find( 'update' ).each( function() {
157       // grab key info from xml
158       var update = {
159         timestamp: $( this ).find( 'timestamp' ).first().text(),
160         firstName: $( this ).find( 'first-name' ).first().text(),
161         lastName: $( this ).find( 'last-name' ).first().text(),
162         photo: $( this ).find( 'person' ).first().find( 'picture-url' ).text(),
163         profileUrl: $( this ).find( 'site-standard-profile-request' ).first().find( 'url' ).text(),
164         status: $( this ).find( 'current-status' ).text(),
165         comments: $( this ).find( 'update-comments' ),
166         updateKey: $( this ).find( 'update-key' ).text()
167       };
168 
169       // handle comments
170       var commentClass = 'no-comments',
171           commentHtml  = '';
172       if( update.comments.length !== 0 ) {
173         // set text for 'X comments' link
174         update.commentTotal = update.comments.attr( 'total' );
175         update.commentTotal = update.commentTotal + ( ( update.commentTotal == 1 ) ? ' comment' : ' comments' );
176         commentClass = 'has-comments';
177 
178         // gather comment data
179         commentHtml += '<ul>';
180         $( this ).find( 'update-comment' ).each( function() {
181           var comment = {
182             timestamp: $( this ).find( 'timestamp' ).first().text(),
183             firstName: $( this ).find( 'first-name' ).first().text(),
184             lastName: $( this ).find( 'last-name' ).first().text(),
185             photo: $( this ).find( 'person' ).first().find( 'picture-url' ).text(),
186             profileUrl: $( this ).find( 'site-standard-profile-request' ).first().find( 'url' ).text(),
187             comment: $( this ).find( 'comment' ).text()
188           };
189 
190           // handle no photo case
191           if( comment.photo === '' ) comment.photo = DEFAULT_USER_PHOTO;
192 
193           // convert timestamp to date
194           var date = new Date();
195           date.setTime( comment.timestamp );
196 
197           // inject data into markup
198           commentHtml += [
199             '<li>',
200               '<a href="', comment.profileUrl, '" title="View profile" target="_blank">',
201                 '<img src="', comment.photo, '" class="photo" alt="">',
202               '</a> ',
203               '<div class="info">',
204                 '<a href="', comment.profileUrl, '" title="View profile" target="_blank">',
205                   '<strong>', comment.firstName, ' ', comment.lastName, '</strong>',
206                 '</a> ',
207                 '<span class="text">', $.link( escapeHTML( comment.comment ) ), '</span>',
208                 '<span class="meta">',
209                   '<time class="timeago" datetime="', ISODateString(date), '">', $.timeago( date ), '</time>',
210                 '</span>',
211               '</div>',
212             '</li>'
213           ].join('');            
214         } );
215         commentHtml += '</ul>';
216       }
217 
218       // handle no photo case
219       if( update.photo === '' ) update.photo = DEFAULT_USER_PHOTO;
220 
221       // convert timestamp to date
222       var date = new Date();
223       date.setTime( update.timestamp );
224 
225       // inject data into markup
226       html += [
227         '<li class="STAT ', commentClass, '">',
228           '<a href="', update.profileUrl, '" title="View profile" target="_blank">',
229             '<img src="', update.photo, '" class="photo" alt="">',
230           '</a>',
231           '<div class="info">',
232             '<a href="', update.profileUrl, '" title="View profile" target="_blank">',
233               '<strong>', update.firstName, ' ', update.lastName, '</strong>',
234             '</a> ',
235             '<span class="text">', $.link( escapeHTML( update.status ) ), '</span>',
236             '<span class="meta">',
237               '<time class="timeago" datetime="', ISODateString(date), '">', $.timeago( date ), '</time>',
238               '<span class="comments total-link">',
239                 '<a href="#comment-detail-', update.updateKey, '" class="comment-toggle">', update.commentTotal, '</a>',
240               '</span>',
241               '<span class="comments simple-link">',
242                 '<a href="#comment-detail-', update.updateKey, '" class="comment-toggle">Comment</a>',
243               '</span>',
244             '</span>',
245           '</div>',
246           '<div class="comment-detail" id="comment-detail-', update.updateKey,'">',
247             commentHtml,
248             '<form action="#" id="', update.updateKey, '">',
249               '<img src="', localStorage.userPhoto, '" class="photo" alt="">',
250               '<textarea></textarea>',
251               '<input type="submit" name="comment-', update.timestamp, '" value="Comment">',
252             '</form>',
253           '</div>',
254         '</li>'
255       ].join('');
256     } );
257     html += '</ul>';
258     document.getElementById( 'updates' ).innerHTML = html;
259 
260     // update timestamps
261     $( 'time.timeago' ).timeago();
262   }
263   
264   // initialize event handlers for links/actions
265   initHandlers();
266 
267   // reset badge
268   chrome.browserAction.setBadgeText({
269     text: ''
270   });
271   
272   // update lastCheckTime
273   localStorage.lastCheckTime = new Date().getTime();
274 }

276 /**
277  * Processes response to API request to post status update, displaying any
278  * error/success messaging and adding to feed.
279  * @param {String} The text response of the API XHR call.
280  * @param {XMLHttpRequest} resp The XMLHttpRequest object which was used to 
281  *     send the request.
282  * @param {Object} bonus Object including text of posted status so it can be
283  *     displayed in the feed.
284  */
285 function processStatusUpdate( resp, xhr, bonus ) {
286   if( xhr.status == 204 ) {
287     showAlert( 'Your status has been posted!', 'success' );
288     $( '#status' ).val( '' );
289     $( '#status' ).blur();
290     addToFeed( bonus[ 0 ] );
291   } else {
292     showAlert( 'Uh-oh. Something went wrong. Please try again.', 'error' );
293     console.log( xhr );
294   }
295 }

297 /**
298  * Processes response to API request to post comment, displaying any
299  * error/success messaging and adding to feed.
300  * @param {String} The text response of the API XHR call.
301  * @param {XMLHttpRequest} resp The XMLHttpRequest object which was used to 
302  *     send the request.
303  * @param {Object} bonus Object including text of posted comment so it can be
304  *     displayed in the feed.
305  */
306 function processComment( resp, xhr, bonus ) {
307   if( xhr.status == 201 ) {
308     showAlert( 'Your comment has been posted!', 'success' );
309     $( 'ul textarea' ).val( '' );
310     $( 'ul textarea' ).blur();
311     addToDiscussion( bonus[ 0 ], bonus[ 1 ] );
312   } else {
313     showAlert( 'Uh-oh. Something went wrong. Please try again.', 'error' );
314     console.log( xhr );
315   }
316 }

318 /**
319  * Add event handlers for submitting status, submitting comment, and
320  * toggling display of comments.
321  */
322 function initHandlers() {
323   // add handler for submitting status
324   $( '#status-form' ).submit( function() {
325     var status = $( '#status' ).val();
326     
327     if( status === '' ) {
328       showAlert( 'A blank status isn\'t that interesting.', 'error' );
329     } else {
330       oauth.authorize( function() {
331         postStatusUpdate( status );
332       } );
333     }
334     return false;
335   } );
336   
337   // add handler for submitting comments
338   $( 'ul' ).submit( function( event ) {
339     var elem = event.target;
340     var updateKey = $( elem ).attr( 'id' );
341     var comment = $( elem ).find( 'textarea' ).val();
342     
343     if( comment === '' ) {
344       showAlert( 'A blank comment isn\'t that interesting.', 'error' );
345     } else {
346       oauth.authorize( function() {
347         postComment( comment, updateKey );
348       } );
349     }
350     return false;
351   } );
352   
353   // add handler for toggling comments
354   $( 'ul' ).click( function( event ) {
355     var elem = event.target;
356     if( elem.nodeName.toUpperCase() === 'A' && elem.className === 'comment-toggle' ) {
357       var updateKey = elem.hash.split( 'comment-detail-' )[ 1 ];
358       var commentDiv = document.getElementById( 'comment-detail-' + updateKey );
359       var textarea = commentDiv.getElementsByTagName( 'textarea' )[ 0 ];
360       $( commentDiv ).slideToggle( 'fast' );
361       $( textarea ).focus();
362       return false;
363     }
364   } );
365   
366   // add target=blank to links inside status/comments
367   $( '.text a' ).click( function() {
368     $( this ).attr( 'target', '_blank' );
369   } );
370 }

372 /**
373  * Adds new status update to feed.
374  * @param {String} txt The text of the update we want to post.
375  */
376 function addToFeed( txt ) {
377   var date = new Date();
378   var html = [
379     '<li class="STAT no-comments">',
380       '<a href="', localStorage.profileUrl, '" title="View profile">',
381         '<img src="', localStorage.userPhoto, '" class="photo" alt="">',

383       '<div class="info">',

387         '<span class="text">', $.link( escapeHTML( txt ) ), '</span>',
388         '<span class="meta">',
389           '<time class="timeago" datetime="', ISODateString(date), '">', $.timeago( date ), '</time>',
390         '</span>',
391       '</div>',
392     '</li>'
393   ].join('');
394   $( html ).insertBefore( '#feed > li:first-child' );
395   $( '.loading' ).hide();  // in case this is the first update
396 }

398 /**
399  * Adds new comment to feed.
400  * @param {String} txt The text of the comment we want to post.
401  * @param {String} key Unique identifier for the update we are attaching this
402  *     comment to.
403  */
404 function addToDiscussion( txt, key ) {
405   var date = new Date();
406   var html = [
407     '<li>',
408       '<a href="', localStorage.profileUrl, '" title="View profile">',
409         '<img src="', localStorage.userPhoto, '" class="photo" alt="">',
410       '</a> ',
411       '<div class="info">',
412         '<a href="', localStorage.profileUrl, '" title="View profile">',
413           '<strong>', localStorage.firstName, ' ', localStorage.lastName, '</strong>',
414         '</a> ',
415         '<span class="text">', $.link( escapeHTML( txt ) ), '</span>',
416         '<span class="meta">',
417           '<time class="timeago" datetime="', ISODateString(date), '">', $.timeago( date ), '</time>',
418         '</span>',
419       '</div>',
420     '</li>'
421   ].join('');
422   
423   // add to proper comment list (and create list if this is the first one)
424   var commentDiv = document.getElementById( 'comment-detail-' + key );
425   var commentList = commentDiv.getElementsByTagName( 'ul' )[ 0 ];
426   if( commentList ) {
427     html = domify( html );
428     commentList.appendChild( html );
429   } else {
430     html = '<ul>' + html + '</ul>';
431     html = domify( html );
432     commentDiv.insertBefore( html, commentDiv.firstChild );
433     commentList = commentDiv.getElementsByTagName( 'ul' )[ 0 ];
434   }
435   
436   // update comment totals in links
437   var totalComments = commentList.getElementsByTagName( 'li' ).length;
438   $( commentDiv ).parent().find( '.comments a ').each( function() {
439     $( this ).html( totalComments + ( ( totalComments == 1 ) ? ' comment' : ' comments' ) );
440   } );
441   $( commentDiv ).parent().addClass( 'has-comments' );
442 }

444 /**
445  * Display error/success messages.
446  * @param {String} msg The message text.
447  * @param {String} type The type of message ('error' or 'success').
448  */
449 function showAlert( msg, type ) {
450   $( '#alert' ).removeClass(); // wipe the slate clean
451   $( '#alert' ).html( msg );
452   $( '#alert' ).addClass( type );
453   $( '#alert' ).slideDown( 'fast', function() {
454     window.setTimeout( function() {
455       $( '#alert ').slideUp( 'fast' );
456     }, 3000 );
457   } );
458 }

460 /**
461  * Escape HTML in user-generated content (updates, comments).
462  * @param {String} str The user-generated content.
463  */
464 function escapeHTML( str ) {
465   return str.replace(/&/g, "&").replace(/</g, "<").replace(/>/g, ">");
466 }

468 /**
469  * Inject content into a list (either the feed or comment list).
470  * @param {String} str The content to add.
471  */
472 function domify( str ) {
473   var ul = document.createElement( 'ul' );
474   ul.innerHTML = str;
475   return ul.firstChild;
476 }

478 /** 
479  * Convert dates into ISO 8601 format (for timeago plugin)
480  * @param {Date} d A Javascript Date object to convert.
481  */
482 function ISODateString( d ) {
483   function pad( n ) {
484     return n < 10 ? '0' + n : n;
485   }
486   
487   return d.getUTCFullYear() + '-' +
488     pad( d.getUTCMonth() + 1 ) + '-' +
489     pad( d.getUTCDate() ) + 'T' +
490     pad( d.getUTCHours() ) + ':' +
491     pad( d.getUTCMinutes() ) + ':' +
492     pad( d.getUTCSeconds() ) + 'Z';
493 }
